## Lifecycle
```ts
defineField()
  .component({
    id: 'email',
    component: 'input',
  })
  .lifecycle({
    onBeforeInit() {
      console.log('onBeforeInit')
    },
    onInit() {
      console.log('onInit')
    },
    onDestroy() {
      console.log('onDestroy')
    },
    onDisabled(state) {
      console.log('onDisabled', state)
    },
    onHidden(state) {
      console.log('onHidden', state)
    },
    onMounted() {
      console.log('onMounted')
    },
    onUnmounted() {
      console.log('onUnmounted')
    },
    onValidate() {
      console.log('onValidate')
    }
  })
  .build()
```
